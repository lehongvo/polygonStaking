Vùng hồi phục: ≤ 60-70% HRmax

Có ít nhất 2 khoảng cường độ cao

Tổng thời gian cao đạt ngưỡng (≥ 60-120 giây)

Hoàn thành = Đạt số bước MỤC TIÊU + Thực hiện HIIT hợp lệ

Light HIIT: cho người mới
Standard HIIT: mặc định

func fetchHeartRateData(startDate: Date, endDate: Date) {
    let heartRateType = HKQuantityType.quantityType(forIdentifier: .heartRate)!
    
    let predicate = HKQuery.predicateForSamples(
        withStart: startDate,
        end: endDate,
        options: .strictStartDate
    )
    
    let query = HKSampleQuery(
        sampleType: heartRateType,
        predicate: predicate,
        limit: HKObjectQueryNoLimit,
        sortDescriptors: nil
    ) { query, results, error in
        guard let samples = results as? [HKQuantitySample] else { return }
        
        for sample in samples {
            let bpm = sample.quantity.doubleValue(for: HKUnit.count().unitDivided(by: .minute()))
            let timestamp = sample.startDate
            
            print("HR: \(bpm) bpm at \(timestamp)")
            // Gửi lên backend
        }
    }
    
    healthStore.execute(query)
}
Bước 3: Đọc số bước

func fetchSteps(for date: Date) {
    let stepsType = HKQuantityType.quantityType(forIdentifier: .stepCount)!
    
    let calendar = Calendar.current
    let startOfDay = calendar.startOfDay(for: date)
    let endOfDay = calendar.date(byAdding: .day, value: 1, to: startOfDay)!
    
    let predicate = HKQuery.predicateForSamples(
        withStart: startOfDay,
        end: endOfDay,
        options: .strictStartDate
    )
    
    let query = HKStatisticsQuery(
        quantityType: stepsType,
        quantitySamplePredicate: predicate,
        options: .cumulativeSum
    ) { query, result, error in
        guard let result = result, let sum = result.sumQuantity() else { return }
        
        let steps = sum.doubleValue(for: HKUnit.count())
        print("Total steps: \(steps)")
    }
    
    healthStore.execute(query)
}

## **5. Chiến lược triển khai cho Exercise Supplement**

### **Phương án 1: Đọc real-time (khuyến nghị)**
```
App → Định kỳ 5-10s đọc HR từ HealthKit/Health Connect
    → Lưu vào local database
    → Cuối ngày gửi lên backend để tính HIIT
```

### **Phương án 2: Đọc cuối ngày**
```
23:00 mỗi ngày → App đọc toàn bộ HR + Steps trong ngày
                → Gửi lên backend
                → Backend tính HIIT và cập nhật DailySuccess
```

### **Phương án 3: Background sync**
```
iOS: HealthKit Background Delivery
Android: WorkManager với PeriodicWorkRequest
→ Tự động sync khi có dữ liệu mới
```

4. Logic tính HIIT (Pseudo-code chi tiết)
python
# ========================================
# HIIT Detection Algorithm
# ========================================

class HIITProfile:
    def __init__(self, name, high_threshold, recovery_threshold, 
                 min_high_duration, min_high_count, min_high_total,
                 work_rest_ratio=None):
        self.name = name
        self.high_threshold = high_threshold      # % MHR
        self.recovery_threshold = recovery_threshold
        self.min_high_duration = min_high_duration  # giây
        self.min_high_count = min_high_count
        self.min_high_total = min_high_total       # giây
        self.work_rest_ratio = work_rest_ratio     # ví dụ: (1, 2) = 1:2

# Định nghĩa profiles
PROFILES = {
    'light': HIITProfile('Light HIIT', 0.80, 0.65, 20, 2, 90),
    'standard': HIITProfile('Standard HIIT', 0.85, 0.65, 30, 3, 120),
    'athletic': HIITProfile('Athletic HIIT', 0.90, 0.65, 30, 5, 300, (1, 2)),
    'safety': HIITProfile('Safety HIIT', 0.75, 0.60, 15, 2, 60)
}


def calculate_MHR(age):
    """Tính nhịp tim tối đa"""
    return 220 - age


def detect_hiit_intervals(hr_data, profile, age):
    """
    Phát hiện các hiệp HIIT từ dữ liệu nhịp tim
    
    Args:
        hr_data: List[{timestamp, heart_rate}]
        profile: HIITProfile
        age: int
    
    Returns:
        {
            'success': bool,
            'high_intervals': List[{start, end, duration}],
            'total_high_duration': int,
            'stats': dict
        }
    """
    
    MHR = calculate_MHR(age)
    high_hr_threshold = MHR * profile.high_threshold
    recovery_hr_threshold = MHR * profile.recovery_threshold
    
    # Sắp xếp theo thời gian
    hr_data = sorted(hr_data, key=lambda x: x['timestamp'])
    
    # Phát hiện các khoảng cường độ cao
    high_intervals = []
    current_high = None
    last_state = None  # 'high', 'recovery', None
    
    for i, point in enumerate(hr_data):
        hr = point['heart_rate']
        timestamp = point['timestamp']
        
        # Xác định trạng thái hiện tại
        if hr >= high_hr_threshold:
            current_state = 'high'
        elif hr <= recovery_hr_threshold:
            current_state = 'recovery'
        else:
            current_state = 'transition'  # vùng trung gian
        
        # Bắt đầu khoảng HIGH
        if current_state == 'high' and last_state != 'high':
            current_high = {
                'start': timestamp,
                'end': timestamp,
                'duration': 0
            }
        
        # Đang trong khoảng HIGH
        elif current_state == 'high' and last_state == 'high':
            current_high['end'] = timestamp
            if i > 0:
                time_diff = (timestamp - hr_data[i-1]['timestamp']).total_seconds()
                current_high['duration'] += time_diff
        
        # Kết thúc khoảng HIGH
        elif current_state != 'high' and last_state == 'high':
            # Chỉ ghi nhận nếu đủ dài
            if current_high['duration'] >= profile.min_high_duration:
                high_intervals.append(current_high)
            current_high = None
        
        last_state = current_state
    
    # Xử lý hiệp cuối nếu đang HIGH
    if current_high and current_high['duration'] >= profile.min_high_duration:
        high_intervals.append(current_high)
    
    # Kiểm tra có hồi phục giữa các HIGH không
    valid_intervals = []
    for i in range(len(high_intervals)):
        if i == 0:
            valid_intervals.append(high_intervals[i])
        else:
            # Kiểm tra có recovery giữa high_intervals[i-1] và high_intervals[i]
            recovery_start = high_intervals[i-1]['end']
            recovery_end = high_intervals[i]['start']
            
            # Lấy HR trong khoảng recovery
            recovery_hr = [
                p['heart_rate'] for p in hr_data
                if recovery_start <= p['timestamp'] <= recovery_end
            ]
            
            # Nếu có ít nhất 1 điểm HR ≤ recovery_threshold
            if recovery_hr and min(recovery_hr) <= recovery_hr_threshold:
                valid_intervals.append(high_intervals[i])
    
    # Tính tổng thời gian HIGH
    total_high_duration = sum(interval['duration'] for interval in valid_intervals)
    
    # Kiểm tra điều kiện HIIT
    hiit_success = (
        len(valid_intervals) >= profile.min_high_count and
        total_high_duration >= profile.min_high_total
    )
    
    return {
        'success': hiit_success,
        'high_intervals': valid_intervals,
        'total_high_duration': total_high_duration,
        'stats': {
            'high_count': len(valid_intervals),
            'avg_high_duration': total_high_duration / len(valid_intervals) if valid_intervals else 0,
            'MHR': MHR,
            'high_threshold_bpm': high_hr_threshold,
            'recovery_threshold_bpm': recovery_hr_threshold
        }
    }


def check_daily_completion(steps, hr_data, step_goal, profile_name, age):
    """
    Kiểm tra hoàn thành trong ngày
    """
    # Kiểm tra số bước
    step_success = steps >= step_goal
    
    # Kiểm tra HIIT
    profile = PROFILES[profile_name]
    hiit_result = detect_hiit_intervals(hr_data, profile, age)
    hiit_success = hiit_result['success']
    
    # Kết luận
    daily_success = step_success and hiit_success
    
    return {
        'daily_success': daily_success,
        'step_success': step_success,
        'hiit_success': hiit_success,
        'steps': steps,
        'step_goal': step_goal,
        'hiit_details': hiit_result
    }

Age (for HRmax Calculation)
Currently, our challenge app stores the user's birthday, which allows us to calculate their age. However, in cases where the user has not provided their birthday (birthday = null), we propose using a default HRmax value of 110 bpm.

Rationale:

HRmax = 110 bpm is a conservative baseline that ensures safety for users of unknown age
This value is intentionally set lower than typical calculated HRmax to avoid health risks
It represents an idealized minimum threshold suitable for general population screening
Hoặc nếu muốn rõ ràng hơn về logic:

Age Handling for HRmax Calculation
Current Implementation:

The app stores user birthday in the challenge database
Age is derived from birthday to calculate HRmax using the formula: HRmax = 220 - age
Fallback for Missing Data:

When birthday = null, we propose using a default HRmax of 110 bpm (not calculated from age)
This is a conservative, safe baseline value suitable for users with unknown age profiles
Why 110 bpm?

Ensures user safety by avoiding overly high intensity targets
Represents a practical lower bound for general population health screening
Can be updated later when user provides their birthday

def check_daily_success(user_data, challenge_config):
    """
    Challenge có 2 cách hoàn thành:
    Option 1: Steps + Speed
    Option 2: HIIT
    """
    
    # Option 1: Steps + Speed requirement
    option_1_success = False
    if challenge_config.has_step_requirement:
        step_ok = user_data.steps >= challenge_config.step_goal
        speed_ok = user_data.avg_speed >= challenge_config.speed_requirement
        option_1_success = step_ok and speed_ok
    
    # Option 2: HIIT requirement  
    option_2_success = False
    if challenge_config.has_hiit_requirement:
        hiit_result = evaluate_hiit(user_data.hr_data, user_data.age)
        option_2_success = hiit_result['hiit_achieved']
    
    # Daily success nếu đạt ÍT NHẤT 1 trong 2 options
    daily_success = option_1_success or option_2_success
    
    return {
        'daily_success': daily_success,
        'option_1_success': option_1_success,  # Steps + Speed
        'option_2_success': option_2_success,  # HIIT
        'completion_method': 'steps_speed' if option_1_success else 'hiit' if option_2_success else 'none'
    }

function getHIITState(currentHR, HRmax) {
  const zone = calculateZone(currentHR, HRmax);
  
  if (zone >= 4) return "High Intensity";
  if (zone === 2) return "Recovery";
  if (zone === 1) return "Ready";
  return "Idle";
}

function getStateIcon(state) {
  const icons = {
    "High Intensity": "🔥",
    "Recovery": "💚",
    "Ready": "🟢",
    "Idle": "⚪"
  };
  return icons[state] || "⚪";
}

class ChallengeConfig:
    def __init__(self):
        self.step_goal = None      # None = not selected
        self.speed_requirement = None  # None = not selected
        self.hiit_required = False  # False = not selected

def check_daily_success(user_data, challenge_config):
    """
    Completion = ALL selected conditions must be met
    """
    results = []
    
    # Check Steps (if selected)
    if challenge_config.step_goal is not None:
        step_success = user_data.steps >= challenge_config.step_goal
        results.append(step_success)
    
    # Check Speed (if selected)
    if challenge_config.speed_requirement is not None:
        speed_success = user_data.avg_speed >= challenge_config.speed_requirement
        results.append(speed_success)
    
    # Check HIIT (if selected)
    if challenge_config.hiit_required:
        hiit_result = evaluate_hiit(user_data.hr_data, user_data.age)
        results.append(hiit_result['hiit_achieved'])
    
    # ALL selected conditions must be TRUE
    if len(results) == 0:
        return False  # No conditions selected (invalid)
    
    daily_success = all(results)  # AND logic
    
    return {
        'daily_success': daily_success,
        'conditions_checked': len(results),
        'conditions_met': sum(results)
    }

function calculateRounds(hrData, age, config) {
  const HRmax = 220 - age;
  const zone4Min = HRmax * 0.80;
  const zone2Max = HRmax * 0.70;
  
  const validRounds = [];
  let currentRound = null;
  
  for (let i = 0; i < hrData.length; i++) {
    const hr = hrData[i].heartRate;
    
    // Bắt đầu round mới
    if (hr >= zone4Min && !currentRound) {
      currentRound = {
        startTime: hrData[i].timestamp,
        duration: 0,
        maxHR: hr
      };
    }
    
    // Đang trong round
    if (currentRound && hr >= zone4Min) {
      currentRound.duration += timeDiff;
      currentRound.maxHR = Math.max(currentRound.maxHR, hr);
    }
    
    // Kết thúc round (HR giảm)
    if (currentRound && hr < zone4Min) {
      // Kiểm tra round có hợp lệ không
      if (currentRound.duration >= 20) {
        // Kiểm tra có recovery về Zone 2 không
        const hasRecovery = checkRecovery(hrData, i, zone2Max);
        
        if (hasRecovery) {
          validRounds.push(currentRound);
        }
      }
      currentRound = null;
    }
  }
  
  return {
    current: validRounds.length,
    required: config.minRounds,
    display: `${validRounds.length} / ${config.minRounds}`
  };
}

function calculateZone45Time(hrData, age, config) {
  const HRmax = 220 - age;
  const zone4Min = HRmax * 0.80; // 80% = bắt đầu Zone 4
  
  let totalZone45Time = 0;
  
  for (let i = 1; i < hrData.length; i++) {
    const currentHR = hrData[i].heartRate;
    const prevTime = hrData[i - 1].timestamp;
    const currentTime = hrData[i].timestamp;
    
    // Nếu HR ở Zone 4-5
    if (currentHR >= zone4Min) {
      // Tính duration giữa 2 samples
      const duration = (currentTime - prevTime) / 1000; // convert to seconds
      totalZone45Time += duration;
    }
  }
  
  return {
    current: Math.round(totalZone45Time),
    required: config.minZone45Time,
    display: `${Math.round(totalZone45Time)}s / ${config.minZone45Time}s`,
    percentage: (totalZone45Time / config.minZone45Time) * 100
  };
}

// Ví dụ sử dụng:
const result = calculateZone45Time(hrData, 30, { minZone45Time: 90 });
console.log(result.display);     // "75s / 90s"
console.log(result.percentage);  // 83.33%

function evaluateDailyHIIT(userData, challengeConfig) {
  const {
    min_zone45_time_per_day,
    min_rounds_per_day,
    min_interval_duration,
    recovery_required
  } = challengeConfig;
  
  // 1. Tính rounds hợp lệ
  const validRounds = calculateValidRounds(
    userData.hr_data,
    userData.age,
    min_interval_duration,
    recovery_required
  );
  
  // 2. Tính tổng thời gian Zone 4-5
  const totalZone45Time = calculateZone45Time(
    userData.hr_data,
    userData.age
  );
  
  // 3. Kiểm tra điều kiện
  const roundsAchieved = validRounds.length >= min_rounds_per_day;
  const timeAchieved = totalZone45Time >= min_zone45_time_per_day;
  
  // 4. Kết luận ngày
  const dailySuccess = roundsAchieved && timeAchieved;
  
  return {
    daily_success: dailySuccess,
    rounds: {
      current: validRounds.length,
      required: min_rounds_per_day,
      achieved: roundsAchieved
    },
    zone45_time: {
      current: totalZone45Time,
      required: min_zone45_time_per_day,
      achieved: timeAchieved
    }
  };
}

  return {
    challenge_success: challengeSuccess,
    success_days: {
      current: successDays,
      required: required_success_days
    }
  };
}

❌ HIIT: Không có

Thế khi nào biết nằm trong trong zone 1-> 5

Feb 10
Khi nào biết đang ở Zone nào?
Công thức:
Bước 1: Tính HRmax = 220 - tuổi
Bước 2: Tính % = (nhịp tim hiện tại / HRmax) × 100
Bước 3: So sánh % với bảng zones
Bảng Zones:
Zone	% HRmax
Zone 1	50-60%
Zone 2	60-70%
Zone 3	70-80%
Zone 4	80-90%
Zone 5	90-100%
Ví dụ cụ thể:
User 30 tuổi:
HRmax = 220 - 30 = 190 bpm

Zone 1: 190 × 50-60% = 95-114 bpm
Zone 2: 190 × 60-70% = 114-133 bpm
Zone 3: 190 × 70-80% = 133-152 bpm
Zone 4: 190 × 80-90% = 152-171 bpm
Zone 5: 190 × 90-100% = 171-190 bpm
Nếu nhịp tim hiện tại = 165 bpm:
% = (165 / 190) × 100 = 86.8%

86.8% nằm trong 80-90%
→ Zone 4 → "High Intensity"
Tóm lại:
App nhận HR từ HealthKit/Health Connect
→ Tính % = HR / HRmax × 100
→ So sánh với bảng zones
→ Biết đang ở Zone mấy
